const SHORTCUT_ROW_HEIGHT: f32 = 44.0;
/// Recording mode chord timeout (longer than runtime 500ms — gives user time to press second key).
const RECORDING_CHORD_TIMEOUT: Duration = Duration::from_millis(1000);

/// Shortcut groups for organizing the list.
#[derive(Debug, Clone, Copy)]
pub enum ShortcutGroup {
    Navigation,
    Panels,
    Terminal,
    Canvas,
}

impl ShortcutGroup {
    pub fn all() -> &'static [ShortcutGroup] { &[Navigation, Panels, Terminal, Canvas] }
    pub fn label(&self) -> &'static str { ... }
    pub fn actions(&self) -> &'static [&'static str] {
        // Returns action IDs belonging to this group
        match self {
            Navigation => &["focus_next_panel", "focus_prev_panel", "toggle_sidebar", "open_settings"],
            Panels => &["panel_split_h", "panel_split_v", "panel_close", "toggle_fullscreen"],
            Terminal => &["create_terminal", "terminal_copy", "terminal_paste", "terminal_search", "font_size_up", "font_size_down"],
            Canvas => &["create_canvas"],
        }
    }
}

/// State for recording a new key binding.
#[derive(Debug, Clone)]
pub enum RecordingState {
    /// Not recording.
    Idle,
    /// Waiting for first key combo.
    WaitingFirst { action_id: String, row_index: usize },
    /// First key captured, waiting for possible second key (chord).
    WaitingChord { action_id: String, row_index: usize, first: KeyCombo, started: Instant },
}

/// Notification toast for conflict resolution.
#[derive(Debug, Clone)]
pub struct NotificationToast {
    pub message: String,          // e.g., "Cmd+D removed from Panel Split"
    pub undo_action_id: String,   // action to restore
    pub undo_keys: Vec<KeyCombo>, // binding to restore
    pub shown_at: Instant,
}

const TOAST_DURATION: Duration = Duration::from_secs(3);
const TOAST_WIDTH: f32 = 280.0;
```

**New methods on SettingsState:**

`pub fn shortcut_row_at(&self, x: f32, y: f32, viewport_y: f32) -> Option<(usize, String)>`:
- Calculate y position of each shortcut row in the content area (right of nav column)
- Account for group headers (13px fg_secondary + 8px padding)
- Returns (flat_row_index, action_id) if click is within a row

`pub fn start_recording(&mut self, action_id: String, row_index: usize)`:
- Sets `self.recording = RecordingState::WaitingFirst { action_id, row_index }`

`pub fn feed_recording_key(&mut self, combo: KeyCombo, registry: &mut ShortcutRegistry) -> Option<SettingsShortcutResult>`:
- If WaitingFirst: set state to WaitingChord { first: combo, started: Instant::now() }. Return None (waiting for possible chord).
- If WaitingChord: full chord captured (first + combo). Call `registry.rebind(action_id, vec![first, combo])`. Handle result (see below). Return result.
- Escape key during recording: cancel, reset to Idle, return Cancelled.
- Backspace/Delete during recording: clear binding (set to empty vec), return Cleared.

`pub fn check_recording_timeout(&mut self, registry: &mut ShortcutRegistry) -> Option<SettingsShortcutResult>`:
- If WaitingChord and elapsed > RECORDING_CHORD_TIMEOUT: treat first key as single-combo binding. Call `registry.rebind(action_id, vec![first])`. Return result.

`pub fn handle_rebind_result(&mut self, displaced: Option<(String, Vec<KeyCombo>)>, action_id: &str, new_keys: &[KeyCombo])`:
- If displaced is Some: create NotificationToast with message "{key combo} removed from {action name}" per D-16 copywriting.
- Push toast to self.toasts.
- Reset recording to Idle.

`pub fn handle_undo(&mut self, registry: &mut ShortcutRegistry)`:
- Pop the newest toast
- Restore the displaced binding: registry.rebind(toast.undo_action_id, toast.undo_keys)
- Save user shortcuts

`pub fn tick_toasts(&mut self)`:
- Remove toasts older than TOAST_DURATION

**Enum for results:**

    Bound { displaced: Option<(String, Vec<KeyCombo>)> },

    Cleared,
}

   fn save_shortcut_overrides(&self) {
       // Compute sparse overrides: compare current registry bindings against defaults
       let defaults = crate::shortcuts::defaults::default_shortcuts();
       let mut overrides = Vec::new();
       for (action_id, keys) in self.shortcut_registry.all_bindings() {
           let default_binding = defaults.iter().find(|d| d.action == action_id);
           let keys_str: Vec<String> = keys.iter().map(|k| crate::shortcuts::chord::key_combo_to_string(k)).collect();
           let is_default = default_binding.map(|d| d.keys == keys_str).unwrap_or(false);
           if !is_default {
               overrides.push(ShortcutEntry { action: action_id.to_string(), keys: keys_str });
           }
       }
       crate::shortcuts::serialization::save_user_shortcuts(&overrides);
   }
   ```
