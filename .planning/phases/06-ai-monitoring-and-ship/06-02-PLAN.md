    ```rust
    /// Freeze a process and its entire group via SIGSTOP.
    /// Only call with PIDs captured at terminal creation time (T-06-02 security constraint).
    /// Returns Ok(()) on success, Err on failure (ESRCH if process already exited).
    pub fn freeze_process_group(child_pid: u32) -> Result<(), std::io::Error> {
        use libc::{SIGSTOP, getpgid, kill, pid_t};
        let pid = child_pid as pid_t;
        let pgid = unsafe { getpgid(pid) };
        if pgid == -1 {
            return Err(std::io::Error::last_os_error());
        }
        // Negative PID = send to entire process group
        let result = unsafe { kill(-pgid, SIGSTOP) };
        if result == -1 {
            Err(std::io::Error::last_os_error())
        } else {
            Ok(())
        }
    }

    /// Unfreeze a process group via SIGCONT.
    pub fn unfreeze_process_group(child_pid: u32) -> Result<(), std::io::Error> {
        use libc::{SIGCONT, getpgid, kill, pid_t};
        let pid = child_pid as pid_t;
        let pgid = unsafe { getpgid(pid) };
        if pgid == -1 {
            return Err(std::io::Error::last_os_error());
        }
        let result = unsafe { kill(-pgid, SIGCONT) };
        if result == -1 {
            Err(std::io::Error::last_os_error())
        } else {
            Ok(())
        }
    }
    ```

    **3. Implement FreezePanel action in process_action:**
    ```rust
    InputAction::FreezePanel { panel_id } => {
        if let Some(panel) = self.grid_layout.get_panel_mut(panel_id) {
            if panel.frozen { return; } // Already frozen
            
            match panel.panel_type {
                PanelType::Terminal => {
                    // Check exited state first (Pitfall 5: SIGSTOP on exited process)
                    let is_exited = self.terminal_manager.as_ref()
                        .and_then(|tm| tm.get_state(&panel_id))
                        .map(|ts| ts.exited)
                        .unwrap_or(true);
                    
                    if !is_exited {
                        if let Some(child_pid) = panel.child_pid {
                            match monitor::freeze_process_group(child_pid) {
                                Ok(()) => { panel.frozen = true; }
                                Err(e) => {
                                    tracing::warn!("Failed to freeze process {}: {}", child_pid, e);
                                    self.toast_manager.add(
                                        toast::ToastType::Error,
                                        "Could not freeze process".to_string(),
                                        Some(format!("SIGSTOP failed: {}", e)),
                                        None, None,
                                        std::time::Duration::from_secs(5),
                                    );
                                }
                            }
                        }
                    }
                }
                PanelType::Canvas | PanelType::Markdown => {
                    // Webview freeze: set_visible(false) per research Pattern 4
                    if let Some(cm) = &self.canvas_manager {
                        if let Some(wv) = cm.get_webview(&panel_id) {
                            let _ = wv.set_visible(false);
                        }
                    }
                    panel.frozen = true;
                }
                _ => {} // Placeholder panels: no-op
            }
        }
    }
    ```

    **4. Implement UnfreezePanel action in process_action:**

    ```rust
    InputAction::UnfreezePanel { panel_id } => {
        if let Some(panel) = self.grid_layout.get_panel_mut(panel_id) {
            if !panel.frozen { return; }
            
            match panel.panel_type {
                PanelType::Terminal => {
                    if let Some(child_pid) = panel.child_pid {
                        match monitor::unfreeze_process_group(child_pid) {
                            Ok(()) => { panel.frozen = false; }
                            Err(e) => {
                                tracing::warn!("Failed to unfreeze process {}: {}", child_pid, e);
                                // Still unfreeze the panel state (process may have exited while frozen)
                                panel.frozen = false;

                    }
                }
                PanelType::Canvas | PanelType::Markdown => {
                    if let Some(cm) = &self.canvas_manager {

                            let _ = wv.set_visible(true);
                        }
                    }
                    panel.frozen = false;
                }
                _ => { panel.frozen = false; }
            }
        }
    }
    ```
