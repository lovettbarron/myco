   ```rust
   /// Scroll the terminal display by delta lines (positive = scroll up/back).
   /// Per D-11: if in ALT_SCREEN mode, send arrow key sequences instead.
   pub fn scroll(&mut self, delta: i32) {
       let term = self.term.lock();
       let mode = *term.mode();
       drop(term); // Release lock before writing
       
       if mode.contains(TermMode::ALT_SCREEN) {
           // D-11: In alternate screen, send arrow keys to the app
           let key = if delta > 0 { b"\x1b[A" } else { b"\x1b[B" };
           for _ in 0..delta.unsigned_abs() {
               self.write_to_pty(key);
           }
       } else {
           let mut term = self.term.lock();
           term.scroll_display(alacritty_terminal::grid::Scroll::Delta(delta));
           self.scroll_offset = term.grid().display_offset();
       }
   }
   
   /// Jump to the latest output (scroll to bottom). Clears new output indicator.
   pub fn scroll_to_bottom(&mut self) {
       let mut term = self.term.lock();
       term.scroll_display(alacritty_terminal::grid::Scroll::Bottom);
       self.scroll_offset = 0;
       self.has_new_output_while_scrolled = false;
   }
   
   /// Called when new PTY output arrives. Updates the new-output indicator per D-10.
   pub fn on_new_output(&mut self) {
       let term = self.term.lock();
       let offset = term.grid().display_offset();
       drop(term);
       if offset > 0 {
           self.has_new_output_while_scrolled = true;
       }
       self.scroll_offset = offset;
   }
   ```

   ```rust
   /// Convert pixel coordinates to terminal grid Point.
   /// viewport_x/y is the top-left of the terminal content area (below panel title bar).
   pub fn pixel_to_point(
       x: f32, y: f32,
       viewport_x: f32, viewport_y: f32,
       cell_width: f32, cell_height: f32,
       display_offset: usize,
   ) -> Point {
       let col = ((x - viewport_x) / cell_width).max(0.0) as usize;
       let row = ((y - viewport_y) / cell_height).max(0.0) as usize;
   ```

   ```rust
   pub fn trigger_copy_flash(&mut self) {
       let term = self.term.lock();
       if let Some(sel) = &term.selection {
           self.copy_flash_selection = Some(sel.clone());
       }
       drop(term);
       self.copy_flash_start = Some(std::time::Instant::now());

   /// Check if copy flash is active and return opacity (0.0-1.0).
   /// Returns None if flash has expired (>200ms).
   pub fn copy_flash_opacity(&self) -> Option<f32> {
       if let Some(start) = self.copy_flash_start {
           let elapsed = start.elapsed().as_millis() as f32;
           if elapsed < 200.0 {
               Some(1.0 - (elapsed / 200.0)) // Fade from 1.0 to 0.0 over 200ms
           } else {
               None
           }
       } else {
           None
       }
   }
   
   /// Clear expired flash state.
   pub fn clear_expired_flash(&mut self) {
       if let Some(start) = self.copy_flash_start {
           if start.elapsed().as_millis() > 200 {
               self.copy_flash_start = None;
               self.copy_flash_selection = None;
           }
       }
   }
   ```

   Define search state machine:
   ```rust
   /// Search overlay state (per D-09: Chrome/VS Code style search bar).
   #[derive(Debug)]
   pub enum SearchState {
       /// Search overlay is closed.
       Closed,
       /// Search overlay is open with active query.
       Open {
           /// Current search query text.
           query: String,
           /// Compiled regex for the query (None if query is empty or invalid).
           regex: Option<RegexSearch>,
           /// Total number of matches found.
           match_count: usize,
           /// Index of the currently focused match (0-based).
           current_match: usize,
           /// Positions of all matches (for highlighting).
           match_positions: Vec<SearchMatch>,
       },
   }
   
   /// A single search match with start and end grid positions.
   #[derive(Debug, Clone)]
   pub struct SearchMatch {
       pub start: Point,
       pub end: Point,
   }
   
   impl SearchState {
       pub fn new() -> Self {
           SearchState::Closed
       }
       
       /// Open the search overlay (D-09: Cmd+F).
       pub fn open(&mut self) {
           *self = SearchState::Open {
               query: String::new(),
               regex: None,
               match_count: 0,
               current_match: 0,
               match_positions: Vec::new(),
           };
       }
       
       /// Close the search overlay (D-09: Esc).
       pub fn close(&mut self) {
           *self = SearchState::Closed;
       }
       
       /// Update the search query and re-search (D-09: type to search).
       pub fn update_query(&mut self, term: &mut Term<MycoEventListener>, new_query: &str) {
           if let SearchState::Open { query, regex, match_count, current_match, match_positions } = self {
               *query = new_query.to_string();
               
               if new_query.is_empty() {
                   *regex = None;
                   *match_count = 0;
                   *current_match = 0;
                   match_positions.clear();
                   return;
               }
               
               // Escape the query for literal search (not regex from user perspective)
               let escaped = regex_automata::util::syntax::escape(new_query);
               match RegexSearch::new(&escaped) {
                   Ok(search) => {
                       // Find all matches
                       // Use term.search_next() repeatedly to collect matches
                       let mut matches = Vec::new();
                       // Start from the beginning of scrollback
                       let start = Point::new(
                           alacritty_terminal::index::Line(-(term.grid().history_size() as i32)),
                           alacritty_terminal::index::Column(0),
                       );
                       
                       // Collect matches by iterating with search_next
                       // Note: This is a simplified approach -- for performance,
                       // limit to a reasonable number of matches (e.g., 1000)
                       let mut current_pos = start;
                       for _ in 0..1000 {
                           if let Some(m) = term.search_next(
                               &search,
                               current_pos,
                               alacritty_terminal::term::search::Direction::Right,
                               alacritty_terminal::selection::Side::Left,
                               None,
                           ) {
                               let match_end = m.end();
                               matches.push(SearchMatch {
                                   start: m.start(),
                                   end: match_end,
                               });
                               // Move past this match
                               current_pos = Point::new(
                                   match_end.line,
                                   match_end.column + 1,
                               );
                           } else {
                               break;
                           }
                       }
                       
                       *match_count = matches.len();
                       *current_match = if matches.is_empty() { 0 } else { matches.len() - 1 };
                       *match_positions = matches;
                       *regex = Some(search);
                   }
                   Err(_) => {
                       *regex = None;
                       *match_count = 0;
                       match_positions.clear();
                   }
               }
           }
       }
       
       /// Navigate to next match (D-09: Enter).
       pub fn next_match(&mut self, term: &mut Term<MycoEventListener>) {
           if let SearchState::Open { current_match, match_positions, .. } = self {
               if !match_positions.is_empty() {
                   *current_match = (*current_match + 1) % match_positions.len();
                   // Scroll term to show the current match
                   let match_line = match_positions[*current_match].start.line;
                   // Scroll to make match visible
                   let screen_lines = term.screen_lines() as i32;
                   let display_offset = term.grid().display_offset() as i32;
                   let match_screen_line = match_line.0 + display_offset;
                   if match_screen_line < 0 || match_screen_line >= screen_lines {
                       // Match is off-screen, scroll to center it
                       let target_offset = -(match_line.0) + screen_lines / 2;
                       let current_offset = display_offset;
                       let delta = target_offset - current_offset;
                       term.scroll_display(alacritty_terminal::grid::Scroll::Delta(delta));
                   }
               }
           }
       }
       
       /// Navigate to previous match (D-09: Shift+Enter).
       pub fn prev_match(&mut self, term: &mut Term<MycoEventListener>) {
           if let SearchState::Open { current_match, match_positions, .. } = self {
               if !match_positions.is_empty() {
                   *current_match = if *current_match == 0 {
                       match_positions.len() - 1
                   } else {
                       *current_match - 1
                   };
                   // Scroll to make match visible
                   let match_line = match_positions[*current_match].start.line;
                   let screen_lines = term.screen_lines() as i32;
                   let display_offset = term.grid().display_offset() as i32;
                   let match_screen_line = match_line.0 + display_offset;
                   if match_screen_line < 0 || match_screen_line >= screen_lines {
                       let target_offset = -(match_line.0) + screen_lines / 2;
                       let current_offset = display_offset;
                       let delta = target_offset - current_offset;
                       term.scroll_display(alacritty_terminal::grid::Scroll::Delta(delta));
                   }
               }
           }
       }
       
       /// Whether the search overlay is open.
       pub fn is_open(&self) -> bool {
           matches!(self, SearchState::Open { .. })
       }
       
       /// Get the current query text (empty string if closed).
       pub fn query(&self) -> &str {
           match self {
               SearchState::Open { query, .. } => query,
               SearchState::Closed => "",
           }
       }
       
       /// Get match info for display: "N of M" format.
       pub fn match_info(&self) -> Option<(usize, usize)> {
           match self {
               SearchState::Open { current_match, match_count, .. } => {
                   if *match_count > 0 {
                       Some((*current_match + 1, *match_count))
                   } else {
                       None
                   }
               }
               SearchState::Closed => None,
           }
       }
       
       /// Get all match positions for rendering highlights.
       pub fn match_positions(&self) -> &[SearchMatch] {
           match self {
               SearchState::Open { match_positions, .. } => match_positions,
               SearchState::Closed => &[],
           }
       }
       
       /// Get the current match index.
       pub fn current_match_index(&self) -> usize {
           match self {
               SearchState::Open { current_match, .. } => *current_match,
               SearchState::Closed => 0,
           }
       }
   }
   ```
